5def compute_segmentation_metrics(pred_mask: np.ndarray, true_mask: np.ndarray) -> dict: 

6 """ 

7 Compute segmentation metrics between the predicted and ground truth ROI masks. 

8 

9 Parameters 

10 ---------- 

11 pred_mask : np.ndarray 

12 Predicted binary ROI mask. Expected shape is (n_samples, sequence_length) (or any shape, 

13 as long as it matches true_mask). 

14 true_mask : np.ndarray 

15 Ground truth binary ROI mask. Must have the same shape as pred_mask. 

16 

17 Returns 

18 ------- 

19 metrics : dict 

20 Dictionary with the following keys: 

21 - "precision": The precision (positive predictive value). 

22 - "recall": The recall (sensitivity). 

23 - "f1_score": The harmonic mean of precision and recall. 

24 - "iou": Intersection-over-Union metric. 

25 - "dice": Dice coefficient. 

26 """ 

27 # Flatten the arrays so that each pixel is a sample 

28 pred_flat = pred_mask.flatten() 

29 true_flat = true_mask.flatten() 

30 

31 # Compute precision, recall, and F1 score. 

32 precision, recall, f1, _ = precision_recall_fscore_support(true_flat, pred_flat, average='binary') 

33 

34 # Compute Intersection over Union (IoU) 

35 intersection = np.logical_and(true_flat, pred_flat).sum() 

36 union = np.logical_or(true_flat, pred_flat).sum() 

37 iou = intersection / union if union > 0 else 0.0 

38 

39 # Compute Dice coefficient: (2 * Intersection) / (Total area of both masks) 

40 dice = (2 * intersection) / (true_flat.sum() + pred_flat.sum()) if (true_flat.sum() + pred_flat.sum()) > 0 else 0.0 

41 

42 return { 

43 "precision": precision, 

44 "recall": recall, 

45 "f1_score": f1, 

46 "iou": iou, 

47 "dice": dice 

48 }

51def roi_containment_metric(roi_pred: np.ndarray, roi_gt: np.ndarray) -> dict: 

52 """ 

53 Compute the ROI containment metric, which quantifies the fraction of the predicted 

54 ROI that lies within the ground truth ROI. 

55 

56 This metric is useful when you wish to evaluate whether the predicted ROI regions 

57 are completely within the ground truth regions. It is defined as: 

58 

59 containment_ratio = (# predicted ROI pixels that are within ground truth) / (# predicted ROI pixels) 

60 

61 A complementary misfit ratio is defined as: 

62 

63 misfit_ratio = 1 - containment_ratio 

64 

65 Parameters 

66 ---------- 

67 roi_pred : numpy.ndarray 

68 A binary (0/1) numpy array of predicted ROI mask. Shape can be (H, W) or any 

69 shape as long as it matches roi_gt. 

70 roi_gt : numpy.ndarray 

71 A binary (0/1) numpy array of the ground truth ROI mask, of the same shape as roi_pred. 

72 

73 Returns 

74 ------- 

75 metrics : dict 

76 A dictionary containing: 

77 - 'containment_ratio': float 

78 The fraction of predicted ROI pixels that are within the ground truth ROI. 

79 - 'misfit_ratio': float 

80 The complement of the containment ratio (i.e., fraction of predicted ROI pixels 

81 that are outside the ground truth ROI). 

82 - 'n_pred': int 

83 The total number of predicted ROI pixels. 

84 - 'n_intersection': int 

85 The number of predicted ROI pixels that lie within the ground truth ROI. 

86 

87 Notes 

88 ----- 

89 - If there are no predicted ROI pixels (i.e. roi_pred.sum() == 0), the function returns 

90 a containment ratio of 0 and misfit ratio of 1. 

91 - This metric only considers the predicted ROI pixels. It does not penalize missed ROI pixels 

92 in the ground truth (for that you might combine with recall-based metrics or IoU). 

93 

94 Examples 

95 -------- 

96 >>> roi_pred = np.array([[0, 1, 1], [0, 1, 0]]) 

97 >>> roi_gt = np.array([[0, 1, 0], [0, 1, 0]]) 

98 >>> metrics = roi_containment_metric(roi_pred, roi_gt) 

99 >>> print(metrics['containment_ratio']) 

100 0.6666666666666666 

101 """ 

102 # Ensure inputs are binary arrays 

103 roi_pred = (roi_pred > 0).astype(np.uint8) 

104 roi_gt = (roi_gt > 0).astype(np.uint8) 

105 

106 # Total number of predicted ROI pixels 

107 n_pred = roi_pred.sum() 

108 

109 if n_pred == 0: 

110 # If no ROI is predicted, we define containment as 0 (or you could define it as 1 by convention) 

111 return { 

112 "containment_ratio": 0.0, 

113 "misfit_ratio": 1.0, 

114 "n_pred": 0, 

115 "n_intersection": 0 

116 } 

117 

118 # Intersection of predicted ROI with ground truth ROI 

119 intersection = np.logical_and(roi_pred, roi_gt).astype(np.uint8) 

120 n_intersection = intersection.sum() 

121 

122 # Compute metrics 

123 containment_ratio = n_intersection / n_pred 

124 misfit_ratio = 1 - containment_ratio 

125 

126 return { 

127 "containment_ratio": containment_ratio, 

128 "misfit_ratio": misfit_ratio, 

129 "n_pred": int(n_pred), 

130 "n_intersection": int(n_intersection) 

131 }

134def mc_dropout_prediction( 

135 model: models.Model, 

136 signals: np.ndarray, 

137 num_samples: int = 30) -> tuple[np.ndarray, np.ndarray]: 

138 """ 

139 Perform Monte Carlo (MC) dropout to estimate the mean and uncertainty of ROI predictions. 

140 

141 The function runs multiple stochastic forward passes (with dropout active) and 

142 returns the mean prediction and standard deviation across these passes. 

143 

144 Parameters 

145 ---------- 

146 model : tensorflow.keras.models.Model 

147 A trained Keras model that outputs {'ROI': ...} and contains dropout layers. 

148 signals : np.ndarray 

149 The input data to predict on, with shape (batch_size, sequence_length, 1). 

150 num_samples : int, optional 

151 Number of forward passes through the network. Default is 30. 

152 

153 Returns 

154 ------- 

155 mean_prediction : np.ndarray 

156 The mean of the ROI predictions across `num_samples` forward passes. 

157 Shape (batch_size, sequence_length, 1). 

158 uncertainty : np.ndarray 

159 The standard deviation (std) across the multiple predictions, same shape as 

160 `mean_prediction`. 

161 

162 Notes 

163 ----- 

164 - Dropout is forced to be active by calling the model with `training=True`. 

165 - The `uncertainty` metric is a simple std; you could use alternative uncertainty 

166 measures. 

167 

168 Example 

169 ------- 

170 >>> model = build_ROI_model(sequence_length=128, dropout_rate=0.3) 

171 >>> # Assume model is trained 

172 >>> input_data = np.random.rand(10, 128, 1) 

173 >>> mean_pred, std_pred = mc_dropout_prediction(model, input_data, num_samples=50) 

174 >>> print(mean_pred.shape, std_pred.shape) 

175 (10, 128, 1) (10, 128, 1) 

176 """ 

177 predictions = np.array([ 

178 model(signals, training=True)['ROI'].numpy() for _ in range(num_samples) 

179 ]) 

180 mean_prediction = predictions.mean(axis=0) 

181 uncertainty = predictions.std(axis=0) 

182 return mean_prediction, uncertainty

184def filter_predictions( 

185 model: models.Model, 

186 signals: np.ndarray, 

187 n_samples: int = 30, 

188 threshold: float = 0.9, 

189 std_threshold: float = 0.1) -> np.ndarray: 

190 """ 

191 Estimate a binarized ROI mask using Monte Carlo dropout sampling. 

192 

193 This function repeatedly calls the model in training mode to obtain multiple 

194 stochastic predictions via dropout. It then computes the average (mean) and standard 

195 deviation across these predictions, applies a threshold to create a binary mask, 

196 and optionally returns the standard deviation as a measure of uncertainty. 

197 

198 Parameters 

199 ---------- 

200 model : tensorflow.keras.models.Model 

201 A trained Keras model with dropout layers. Must output a dictionary containing 

202 the key 'ROI'. 

203 signals : np.ndarray 

204 Input tensor of shape (batch_size, sequence_length, 1). The function 

205 repeatedly passes this data through the model in training mode. 

206 n_samples : int, optional 

207 Number of forward passes through the model. Default is 30. 

208 threshold : float, optional 

209 Probability threshold to binarize the mean prediction. Default is 0.9. 

210 

211 Returns 

212 ------- 

213 mean_prediction : np.ndarray 

214 Binarized prediction (0 or 1) based on the threshold. The shape will match 

215 (batch_size, sequence_length, 1). Values below the threshold are set to 0, 

216 above or equal to threshold are set to 1. 

217 

218 Notes 

219 ----- 

220 - Internally, the function also computes the standard deviation (std) across 

221 predictions. You can modify the code to return this `uncertainty` if needed. 

222 - This approach exploits dropout at inference time to estimate model uncertainty. 

223 

224 Examples 

225 -------- 

226 >>> model = build_ROI_model(128, dropout_rate=0.3) 

227 >>> # Assume model is trained... 

228 >>> test_data = np.random.rand(5, 128, 1) 

229 >>> filtered_mask = filter_predictions(model, test_data, n_samples=10, threshold=0.8) 

230 >>> print(filtered_mask.shape) 

231 (5, 128, 1) 

232 """ 

233 predictions = np.array( 

234 [model(signals, training=True)['ROI'].numpy() for _ in range(n_samples)] 

235 ) 

236 

237 mean_prediction = predictions.mean(axis=0) 

238 

239 uncertainty = predictions.std(axis=0) 

240 

241 mean_prediction[mean_prediction < threshold] = 0 

242 mean_prediction[mean_prediction >= threshold] = 1 

243 

244 std_mask = np.zeros_like(uncertainty) 

245 std_mask[uncertainty < std_threshold] = 1 

246 

247 mean_prediction *= std_mask 

248 

249 return mean_prediction.squeeze(), uncertainty.squeeze()
